        def add(key1,key2="",value="")
                if !(key2 == "" and value == "") then
                        add_set key1,key2,value
                elsif /[\w|\W]*?\.yaml/ =~ key1 then
                        add_yaml key1
                elsif /[\w|\W]*?\.txt/ =~ key1 then
                        add_txt key1
                elsif key1.instance_of? String then
                        add_str key1
                end
        end
