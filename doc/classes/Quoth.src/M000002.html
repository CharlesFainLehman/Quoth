        def get(startWord, length)
                ret = startWord + " "
                word = startWord.split
                count = 0
                
                while @corpus.has_key? word and count < length
                        newWord = @corpus[word][rand(@corpus[word].length)]
                        ret << newWord + " " 
                        word = [word[1],newWord]
                        count += 1
                end
                
                ret
        end
