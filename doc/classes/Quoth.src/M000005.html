        def to_s
                ret = ""
                @corpus.each_pair do |key, value|
                        ret << "[#{key[0].to_s},#{key[1].to_s}] => ["
                        for word in value do ret << word + "," end
                        ret.chop!
                        ret << "], "
                end
                ret.chop.chop
        end
